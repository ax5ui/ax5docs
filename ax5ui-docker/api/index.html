Example

var myDocker = new ax5.ui.docker();
myDocker.setConfig({
     target: $('[data-ax5docker="docker1"]'),
     panels: [
         {
             type: "panel",
             name: "panel name",
             color: "#ff3300",
             borderColor: "#000000",
             moduleName: "content",
             moduleState:{
                 data: "data1"
             }
         }
     ]
});
