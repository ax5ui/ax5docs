Split Panel

HTML 
<div data-ax5layout="ax1" data-config="{layout:"split-panel", oriental: "horizontal"}" style="height: 500px;border:1px solid #ccc;">
    <div data-split-panel="{width: "300", minWidth:50}"></div>
    <div data-splitter="{}"></div>
    <div data-split-panel="{width: "*"}">
        <h4>Panel</h4>
    </div>
    <div data-splitter="{}"></div>
    <div data-split-panel="{width: "200", minWidth: 50, maxWidth: 300}"></div>
</div>

Panel

JS 
<script type="text/javascript">
$(document.body).ready(function(){
    $('[data-ax5layout]').ax5layout();
});
</script>
