Basic

Paste the following code into the head section of your site’s HTML.

<link rel="stylesheet" type="text/css" href="src/ax5ui-autocomplete/dist/ax5autocomplete.css">
<script type="text/javascript" src="src/ax5ui-autocomplete/dist/ax5autocomplete.min.js"></script>

    $(document.body).ready(function () {
        $('[data-ax5autocomplete]').ax5autocomplete({
            removeIcon: '<i class="fa fa-times" aria-hidden="true"></i>',
            onSearch: function (callBack) {
                var searchWord = this.searchWord;
                setTimeout(function () {
                    var regExp = new RegExp(searchWord);
                    var myOptions = [];
                    options.forEach(function (n) {
                        if (n.text.match(regExp)) {
                            myOptions.push({
                                value: n.value,
                                text: n.text
                            })
                        }
                    });
                    callBack({
                        options: myOptions
                    });
                }, 150);

            }
        });
    });
